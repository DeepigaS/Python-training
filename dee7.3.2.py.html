def longest_word(input_str): w=input_str.split() longest_word="" longest_length=0 for word in w: if len(word)>longest_length: longest_word=word longest_length=len(word) return longest_word,longest_length input_str=input("Enter a string:") longest_word,length=longest_word(input_str) print("Longest word:",longest_word) print("Length of the longest word:",length)
